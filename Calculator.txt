public class Calculator {
    public int add(int a, int b) {
        int c = a + b;
        return c;
    }

    public int minus(int a, int b) {
        int c = a - b;
        return c;
    }

    public int mult(int a, int b) {
        int c = a * b;
        return c;
    }

    public int div(int a, int b) {
        int c = a / b;
        return c;
    }
}
